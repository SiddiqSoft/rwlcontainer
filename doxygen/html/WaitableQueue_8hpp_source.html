223 auto waitUntilEmpty(std::chrono::milliseconds timeoutDuration = std::chrono::milliseconds(1500)) -> bool
224 {
225 constexpr std::chrono::milliseconds spinInterval {32};
226 std::chrono::milliseconds spinDuration {32};
227
228 auto isEmpty = [&]() -> bool
229 {
230 RLock _ {_containerMutex};
231 return _container.empty();
232 };
233
234 while (!isEmpty() && (spinDuration < timeoutDuration))
235 {
236 // Something is in the queue.. let's spinwait
237 std::this_thread::sleep_for(spinDuration);
238 spinDuration += spinInterval;
239 }
240
241 return isEmpty();
242 }

272 [[nodiscard]] auto tryWaitItem(std::chrono::milliseconds timeoutDuration = std::chrono::milliseconds(100))
273 -> std::optional<StorageType>
274 {
275 // This will wait for the requested interval for an item to be
276 // ready for consumption. If available, we lock and attempt to
277 // pull an item.
278 // It is possible to be signalled and have the item potentially
279 // consumed by another thread (if you have multiple threads against
280 // this object.)
281 if (_signal.try_acquire_for(timeoutDuration))
282 {
283 if (RWLock _ {_containerMutex}; !_container.empty())
284 {
285 // Using this scope guard allows us to minimize object movement
286 siddiqsoft::RunOnEnd scopeCleanup {[&]()
287 {
288 _container.pop_front();
289 _counterRemoves++;
290 }};
291 // Return the object via move since it will be popped by the scope guard
292 return std::make_optional(std::move(_container.front()));
293 }
294 }
295
296 // empty
297 return {};
298 }
